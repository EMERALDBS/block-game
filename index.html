import pygame import random # Initialize pygame pygame.init() # Game Constants WIDTH, HEIGHT = 300, 600 GRID_SIZE = 30 COLUMNS, ROWS = WIDTH // GRID_SIZE, HEIGHT // GRID_SIZE WHITE, BLACK, RED, GREEN, BLUE, CYAN, MAGENTA, YELLOW = (255, 255, 255), (0, 0, 0), (255, 0, 0), (0, 255, 0), (0, 0, 255), (0, 255, 255), (255, 0, 255), (255, 255, 0) COLORS = [RED, GREEN, BLUE, CYAN, MAGENTA, YELLOW] # Tetromino Shapes SHAPES = [ [[1, 1, 1, 1]], # I [[1, 1], [1, 1]], # O [[0, 1, 1], [1, 1, 0]], # S [[1, 1, 0], [0, 1, 1]], # Z [[1, 0, 0], [1, 1, 1]], # L [[0, 0, 1], [1, 1, 1]], # J [[0, 1, 0], [1, 1, 1]] # T ] # Game Variables grid = [[BLACK for _ in range(COLUMNS)] for _ in range(ROWS)] clock = pygame.time.Clock() tetromino = None class Tetromino: def __init__(self): self.shape = random.choice(SHAPES) self.color = random.choice(COLORS) self.x, self.y = COLUMNS // 2 - len(self.shape[0]) // 2, 0 def rotate(self): self.shape = [list(row) for row in zip(*self.shape[::-1])] def draw_grid(): for y in range(ROWS): for x in range(COLUMNS): pygame.draw.rect(screen, grid[y][x], (x * GRID_SIZE, y * GRID_SIZE, GRID_SIZE, GRID_SIZE), 0) pygame.draw.rect(screen, WHITE, (x * GRID_SIZE, y * GRID_SIZE, GRID_SIZE, GRID_SIZE), 1) def draw_tetromino(): for y, row in enumerate(tetromino.shape): for x, cell in enumerate(row): if cell: pygame.draw.rect(screen, tetromino.color, ((tetromino.x + x) * GRID_SIZE, (tetromino.y + y) * GRID_SIZE, GRID_SIZE, GRID_SIZE)) def place_tetromino(): global tetromino for y, row in enumerate(tetromino.shape): for x, cell in enumerate(row): if cell: grid[tetromino.y + y][tetromino.x + x] = tetromino.color tetromino = Tetromino() def move_down(): global tetromino if tetromino.y + len(tetromino.shape) < ROWS - 1: tetromino.y += 1 else: place_tetromino() # Initialize screen screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Tetris") tetromino = Tetromino() running = True while running: screen.fill(BLACK) draw_grid() draw_tetromino() for event in pygame.event.get(): if event.type == pygame.QUIT: running = False elif event.type == pygame.KEYDOWN: if event.key == pygame.K_LEFT and tetromino.x > 0: tetromino.x -= 1 elif event.key == pygame.K_RIGHT and tetromino.x < COLUMNS - len(tetromino.shape[0]): tetromino.x += 1 elif event.key == pygame.K_DOWN: move_down() elif event.key == pygame.K_UP: tetromino.rotate() move_down() pygame.display.flip() clock.tick(5) pygame.quit()
